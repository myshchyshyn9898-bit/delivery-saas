<script>
    let tg = window.Telegram.WebApp;
    let payType = 'cash';
    const bizId = new URLSearchParams(window.location.search).get('biz_id');

    function setPay(type, el) {
        payType = type;
        document.querySelectorAll('.pay-option').forEach(opt => opt.classList.remove('active'));
        el.classList.add('active');
    }

    function sendOrder() {
        const data = {
            action: 'new_order',
            biz_id: bizId,
            address: document.getElementById('address').value,
            amount: document.getElementById('amount').value,
            pay_type: payType,
            phone: document.getElementById('phone').value,
            comment: document.getElementById('comment').value
        };

        if (!data.address || !data.amount) {
            alert("Будь ласка, заповніть адресу та суму!");
            return;
        }

        tg.sendData(json.stringify(data));
        tg.close();
    }
</script>
